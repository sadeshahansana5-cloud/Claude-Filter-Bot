version: '3.8'

services:
  bot:
    build: .
    container_name: sh_telegram_bot
    restart: unless-stopped
    env_file:
      - .env
    volumes:
      - ./sessions:/app/sessions
    depends_on:
      - mongodb
    networks:
      - bot-network

  mongodb:
    image: mongo:7.0
    container_name: sh_mongodb
    restart: unless-stopped
    environment:
      MONGO_INITDB_ROOT_USERNAME: admin
      MONGO_INITDB_ROOT_PASSWORD: your_secure_password
    ports:
      - "27017:27017"
    volumes:
      - mongodb_data:/data/db
    networks:
      - bot-network

volumes:
  mongodb_data:

networks:
  bot-network:
    driver: bridge
